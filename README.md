### 2. Add Two Numbers (JS)
Difficulty: Medium
```bash
var addTwoNumbers = function(l1, l2) {
        let newList = new ListNode(0);
        let head = newList;
        let tenth = 0;
        while (l1 || l2 || tenth) {
                let num = (l1?.val || 0) + (l2?.val || 0) + tenth;
                if(num<=9){
                        head.next = new ListNode(num)
                        tenth = 0
                        }
                else{
                        head.next = new ListNode(num%10)
                        tenth = 1
                        }
            head = head.next;
            l1 = l1?.next;
            l2 = l2?.next;
  }
return newList.next;
};
```
